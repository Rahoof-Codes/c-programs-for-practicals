1.Function Overloading & Inline Functions

Common Errors:

  1. Placement: Forgetting that inline functions must be defined before they are called in main(). 

#include <iostream.h>
#include <conio.h>

inline int square(int x)
{
    return x * x;
}

int add(int a, int b = 10)
{
    return a + b;
}

int add(int a, int b, int c)
{
    return a + b + c;
}

void main()
{
    clrscr();
    cout << "Square of 5: " << square(5) << endl;
    cout << "Add (5 + default): " << add(5) << endl;
    cout << "Add (5 + 6 + 7): " << add(5, 6, 7) << endl;
    getch();
}
